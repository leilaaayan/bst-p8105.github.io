z_scores = function(x) {
  if(!is.numeric(x)) {
    stop("input must be numeric")
  }
  
  if(length(x) < 3){
    stop("input must have at least three numbers")
  }
  z = (x - mean(x)) / sd(x)
  
  return(z)
  
}

sim_mean_sd = function(sample_size, mu=3, sigma=4) {
  
  sim_data = 
  tibble(
    x=rnorm(n=sample_size, mean = mu, sd = sigma)
  )

# compute mean, sd
  sim_data %>%
    summarize(
      mean = mean(x),
      sd= sd(x)
    )
  
}
